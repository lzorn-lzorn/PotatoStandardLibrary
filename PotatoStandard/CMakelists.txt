file(
	GLOB_RECURSE POTATO_STANDARD_PUBLIC_HEADERS
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.h"
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.hh"
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.hpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.hxx"
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.inl"
	"${CMAKE_CURRENT_SOURCE_DIR}/Include/*.ipp"
)

file(
	GLOB_RECURSE POTATO_STANDARD_SOURCES
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/Source/*.c"
	"${CMAKE_CURRENT_SOURCE_DIR}/Source/*.cc"
	"${CMAKE_CURRENT_SOURCE_DIR}/Source/*.cpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/Source/*.cxx"
)


if(POTATO_STANDARD_SOURCES)
	add_library(PotatoStandard)
	set(POTATO_STANDARD_USAGE_SCOPE PUBLIC)

	if(POTATO_STANDARD_SOURCES)
		target_sources(PotatoStandard PRIVATE ${POTATO_STANDARD_SOURCES})
	endif()
else()
	add_library(PotatoStandard INTERFACE)
	set(POTATO_STANDARD_USAGE_SCOPE INTERFACE)
endif()

add_library(PotatoStandard::PotatoStandard ALIAS PotatoStandard)

target_compile_features(PotatoStandard ${POTATO_STANDARD_USAGE_SCOPE} cxx_std_20)
target_include_directories(
	PotatoStandard
	${POTATO_STANDARD_USAGE_SCOPE}
		$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/Include>
		$<INSTALL_INTERFACE:include>
)

set_target_properties(
	PotatoStandard
	PROPERTIES
		EXPORT_NAME PotatoStandard
)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	target_sources(
		PotatoStandard
		${POTATO_STANDARD_USAGE_SCOPE}
			FILE_SET HEADERS
			BASE_DIRS ${CMAKE_CURRENT_SOURCE_DIR}/Include
			FILES ${POTATO_STANDARD_PUBLIC_HEADERS}
	)
endif()

set(POTATO_STANDARD_INSTALL_ARGS EXPORT PotatoStandardTargets)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	list(APPEND POTATO_STANDARD_INSTALL_ARGS FILE_SET HEADERS DESTINATION include)
else()
	install(
		DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/Include/
		DESTINATION include
	)
endif()

install(
	TARGETS PotatoStandard
	${POTATO_STANDARD_INSTALL_ARGS}
)

install(
	EXPORT PotatoStandardTargets
	NAMESPACE PotatoStandard::
	DESTINATION lib/cmake/PotatoStandard
)

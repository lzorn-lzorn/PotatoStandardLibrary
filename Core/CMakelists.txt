if(CMAKE_SOURCE_DIR STREQUAL CMAKE_CURRENT_SOURCE_DIR)
	cmake_minimum_required(VERSION 3.10)
	project(Core VERSION 0.1.0 LANGUAGES CXX)

	set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
	set(CMAKE_CXX_STANDARD 23)
	set(CMAKE_CXX_STANDARD_REQUIRED ON)
	set(CMAKE_CXX_EXTENSIONS OFF)
endif()

include (Memory/CMakeLists.txt)
# add_subdirectory(Memory)

file(
	GLOB_RECURSE POTATO_STANDARD_PUBLIC_HEADERS
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/*.h"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hh"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hxx"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.inl"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.ipp"
)

file(
	GLOB_RECURSE POTATO_STANDARD_SOURCES
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.c"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cxx"
)

if(POTATO_MEMORY_HOOK_SOURCES)
	list(APPEND POTATO_STANDARD_SOURCES ${POTATO_MEMORY_HOOK_SOURCES})
endif()

if(POTATO_STANDARD_SOURCES)
	add_library(Core)
	set(POTATO_STANDARD_USAGE_SCOPE PUBLIC)

	if(POTATO_STANDARD_SOURCES)
		target_sources(Core PRIVATE ${POTATO_STANDARD_SOURCES})
	endif()
else()
	add_library(Core INTERFACE)
	set(POTATO_STANDARD_USAGE_SCOPE INTERFACE)
endif()

add_library(Core::Core ALIAS Core)

target_compile_features(Core ${POTATO_STANDARD_USAGE_SCOPE} cxx_std_23)
target_compile_options(
	Core
	${POTATO_STANDARD_USAGE_SCOPE}
		$<$<CXX_COMPILER_ID:MSVC>:/utf-8>
)

if(POTATO_MEMORY_CAPTURE_STACK)
	set(POTATO_MEMORY_CAPTURE_STACK_VALUE 1)
else()
	set(POTATO_MEMORY_CAPTURE_STACK_VALUE 0)
endif()

if(POTATO_MEMORY_QUARANTINE_RELEASE_ONLY_ON_FLUSH)
	set(POTATO_MEMORY_QUARANTINE_RELEASE_ONLY_ON_FLUSH_VALUE 1)
else()
	set(POTATO_MEMORY_QUARANTINE_RELEASE_ONLY_ON_FLUSH_VALUE 0)
endif()

if(POTATO_MEMORY_LAZY_COMMIT)
	set(POTATO_MEMORY_LAZY_COMMIT_VALUE 1)
else()
	set(POTATO_MEMORY_LAZY_COMMIT_VALUE 0)
endif()

if(POTATO_MEMORY_INTERNAL_BENCH_TIMING)
	set(POTATO_MEMORY_INTERNAL_BENCH_TIMING_VALUE 1)
else()
	set(POTATO_MEMORY_INTERNAL_BENCH_TIMING_VALUE 0)
endif()

target_compile_definitions(
	Core
	${POTATO_STANDARD_USAGE_SCOPE}
		CORE_MEM_CFG_CAPTURE_STACK=${POTATO_MEMORY_CAPTURE_STACK_VALUE}
		CORE_MEM_CFG_QUARANTINE_RELEASE_ONLY_ON_FLUSH=${POTATO_MEMORY_QUARANTINE_RELEASE_ONLY_ON_FLUSH_VALUE}
		CORE_MEM_CFG_LAZY_COMMIT=${POTATO_MEMORY_LAZY_COMMIT_VALUE}
		CORE_MEM_CFG_ENABLE_INTERNAL_BENCH_TIMING=${POTATO_MEMORY_INTERNAL_BENCH_TIMING_VALUE}
)

if(POTATO_MEMORY_DEBUG_GUARDS_MODE STREQUAL "AUTO")
	target_compile_definitions(
		Core
		${POTATO_STANDARD_USAGE_SCOPE}
			"CORE_MEM_CFG_ENABLE_DEBUG_GUARDS=$<IF:$<CONFIG:Debug>,1,0>"
	)
elseif(POTATO_MEMORY_DEBUG_GUARDS_MODE STREQUAL "ON")
	target_compile_definitions(Core ${POTATO_STANDARD_USAGE_SCOPE} CORE_MEM_CFG_ENABLE_DEBUG_GUARDS=1)
elseif(POTATO_MEMORY_DEBUG_GUARDS_MODE STREQUAL "OFF")
	target_compile_definitions(Core ${POTATO_STANDARD_USAGE_SCOPE} CORE_MEM_CFG_ENABLE_DEBUG_GUARDS=0)
else()
	message(FATAL_ERROR "POTATO_MEMORY_DEBUG_GUARDS_MODE must be AUTO, ON, or OFF")
endif()

if(POTATO_MEMORY_UAF_DETECTION_MODE STREQUAL "AUTO")
	target_compile_definitions(
		Core
		${POTATO_STANDARD_USAGE_SCOPE}
			"CORE_MEM_CFG_ENABLE_UAF_DETECTION=$<IF:$<CONFIG:Debug>,1,0>"
	)
elseif(POTATO_MEMORY_UAF_DETECTION_MODE STREQUAL "ON")
	target_compile_definitions(Core ${POTATO_STANDARD_USAGE_SCOPE} CORE_MEM_CFG_ENABLE_UAF_DETECTION=1)
elseif(POTATO_MEMORY_UAF_DETECTION_MODE STREQUAL "OFF")
	target_compile_definitions(Core ${POTATO_STANDARD_USAGE_SCOPE} CORE_MEM_CFG_ENABLE_UAF_DETECTION=0)
else()
	message(FATAL_ERROR "POTATO_MEMORY_UAF_DETECTION_MODE must be AUTO, ON, or OFF")
endif()

target_include_directories(
	Core
	${POTATO_STANDARD_USAGE_SCOPE}
		$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/..>
		$<INSTALL_INTERFACE:include>
)

set_target_properties(
	Core
	PROPERTIES
		EXPORT_NAME Core
)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	target_sources(
		Core
		${POTATO_STANDARD_USAGE_SCOPE}
			FILE_SET HEADERS
			BASE_DIRS ${CMAKE_CURRENT_SOURCE_DIR}/..
			FILES ${POTATO_STANDARD_PUBLIC_HEADERS}
	)
endif()

set(POTATO_STANDARD_INSTALL_ARGS EXPORT CoreTargets)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	list(APPEND POTATO_STANDARD_INSTALL_ARGS FILE_SET HEADERS DESTINATION include)
else()
	install(
		DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
		DESTINATION include
	)
endif()

install(
	TARGETS Core
	${POTATO_STANDARD_INSTALL_ARGS}
)

install(
	EXPORT CoreTargets
	NAMESPACE Core::
	DESTINATION lib/cmake/Core
)

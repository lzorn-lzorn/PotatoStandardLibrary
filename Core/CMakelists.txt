file(
	GLOB_RECURSE POTATO_STANDARD_PUBLIC_HEADERS
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/*.h"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hh"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.hxx"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.inl"
	"${CMAKE_CURRENT_SOURCE_DIR}/*.ipp"
)

file(
	GLOB_RECURSE POTATO_STANDARD_SOURCES
	CONFIGURE_DEPENDS
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.c"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp"
	"${CMAKE_CURRENT_SOURCE_DIR}/src/*.cxx"
)

set(
	POTATO_STANDARD_HOOK_SOURCES
	"${CMAKE_CURRENT_SOURCE_DIR}/Memory/Observability/new_malloc_tracker.cpp"
)
list(APPEND POTATO_STANDARD_SOURCES ${POTATO_STANDARD_HOOK_SOURCES})

if(POTATO_STANDARD_SOURCES)
	add_library(Core)
	set(POTATO_STANDARD_USAGE_SCOPE PUBLIC)

	if(POTATO_STANDARD_SOURCES)
		target_sources(Core PRIVATE ${POTATO_STANDARD_SOURCES})
	endif()
else()
	add_library(Core INTERFACE)
	set(POTATO_STANDARD_USAGE_SCOPE INTERFACE)
endif()

add_library(Core::Core ALIAS Core)

target_compile_features(Core ${POTATO_STANDARD_USAGE_SCOPE} cxx_std_23)
target_include_directories(
	Core
	${POTATO_STANDARD_USAGE_SCOPE}
		$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/..>
		$<INSTALL_INTERFACE:include>
)

set_target_properties(
	Core
	PROPERTIES
		EXPORT_NAME Core
)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	target_sources(
		Core
		${POTATO_STANDARD_USAGE_SCOPE}
			FILE_SET HEADERS
			BASE_DIRS ${CMAKE_CURRENT_SOURCE_DIR}/..
			FILES ${POTATO_STANDARD_PUBLIC_HEADERS}
	)
endif()

set(POTATO_STANDARD_INSTALL_ARGS EXPORT CoreTargets)

if(NOT CMAKE_VERSION VERSION_LESS 3.23 AND POTATO_STANDARD_PUBLIC_HEADERS)
	list(APPEND POTATO_STANDARD_INSTALL_ARGS FILE_SET HEADERS DESTINATION include)
else()
	install(
		DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
		DESTINATION include
	)
endif()

install(
	TARGETS Core
	${POTATO_STANDARD_INSTALL_ARGS}
)

install(
	EXPORT CoreTargets
	NAMESPACE Core::
	DESTINATION lib/cmake/Core
)

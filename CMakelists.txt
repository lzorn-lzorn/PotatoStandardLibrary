cmake_minimum_required(VERSION 3.10)
project(PotatoStandardLibrary VERSION 0.1.0 LANGUAGES CXX)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# 为 MSVC 添加 /utf-8，避免代码页编码警告 C4819
add_compile_options(
    "$<$<CXX_COMPILER_ID:MSVC>:/utf-8>"
)

option(POTATO_STANDARD_BUILD_TESTS "Build tests under Test/." ${PROJECT_IS_TOP_LEVEL})
option(POTATO_STANDARD_BUILD_BENCHMARKS "Build benchmarks under Benchmark/." ${PROJECT_IS_TOP_LEVEL})

add_subdirectory(Core)

if(POTATO_STANDARD_BUILD_BENCHMARKS)
	if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/benchmark/CMakeLists.txt" AND
		EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/PerfBench/CMakeLists.txt")
		set(BENCHMARK_ENABLE_TESTING OFF CACHE BOOL "" FORCE)
		set(BENCHMARK_ENABLE_GTEST_TESTS OFF CACHE BOOL "" FORCE)
		set(BENCHMARK_ENABLE_INSTALL OFF CACHE BOOL "" FORCE)
		set(BENCHMARK_ENABLE_WERROR OFF CACHE BOOL "" FORCE)

		add_subdirectory(benchmark EXCLUDE_FROM_ALL)
		add_subdirectory(PerfBench)
	else()
		message(WARNING "Benchmark sources are incomplete. Skip benchmark targets.")
	endif()
endif()

if(POTATO_STANDARD_BUILD_TESTS)
	include(CTest)

	if(BUILD_TESTING AND EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/Test/CMakeLists.txt")
		add_subdirectory(Test)
	endif()
endif()
